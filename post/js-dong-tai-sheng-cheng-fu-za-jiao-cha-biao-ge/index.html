js动态生成复杂交叉表格

2020-05-12

  • js动态生成复杂交叉表格

    先看看效果

  • 思路

    • 生成包含表头的空表，填入id
    • 填入数值。
    • 填入合计数值。
    • 隐藏整行/整列为空的单元格。

  • 代码

    <html>
  <head>
  	<script src="https://cdn.bootcdn.net/ajax/libs/jquery/3.5.1/jquery.js"></script>
  	<meta charset="utf-8">
  </head>
  <body>
  	<div id="root">
  					
  	</div>
  	
  </body>
  <style type="text/css">
  	table,table tr th,table tr td{
  		border:1px solid;
  	}
  	table{
  		text-align: center;
  		border-collapse: collapse;
  	}
  </style>
  
  <script type="text/javascript">
  $(function(){
  	var d={};
  	var rmData=[
  		['r1a','r1b'],
  		['r2a','r2b'],
  		['r3a','r3b','r3c']
  	];
  	
  	var cmData=[
  		['c1a','c1b','c1c'],
  		['c2a','c2b'],
  		['c3a','c3b','c3c']
  	];
  	d.rmLenArr=[];
  	d.cmLenArr=[];
  	d.data=[
  		['r1a','r2a','r3a','c1a','c2a','c3a',13],
  		['r1a','r2a','r3b','c1b','c2a','c3a',7],
  		['r1a','r2a','r3b','c1c','c2a','c3a',14],
  		['r1a','r2b','r3c','c1a','c2b','c3a',9],
  		['r1a','r2b','r3b','c1b','c2a','c3b',3],
  		['r1a','r2b','r3c','c1c','c2a','c3b',56],
  		['r1b','r2a','r3b','c1a','c2b','c3b',1],
  		['r1b','r2a','r3b','c1b','c2a','c3a',3],
  		['r1b','r2a','r3a','c1c','c2a','c3a',18],
  		['r1b','r2b','r3b','c1a','c2b','c3b',4],
  		['r1b','r2b','r3b','c1b','c2a','c3a',22],
  		['r1b','r2b','r3c','c1c','c2b','c3a',87]
  	];
  	// 错误检查
  	// var map=new Map();
  	// for(var i=0;i<rmData.length;i++){
  	// 	for(var j=0;j<rmData[i].length;j++){
  	// 		if(!map.get(rmData[i][j])){
  	// 			map.set(rmData[i][j],rmData[i][j]);
  	// 		}else{
  	// 			alert('存在同名行/列，数据将不准确!');
  	// 			return;
  	// 		}
  	// 	}
  	// }
  	// for(var i=0;i<cmData.length;i++){
  	// 	for(var j=0;j<cmData[i].length;j++){
  	// 		if(!map.get(cmData[i][j])){
  	// 			map.set(cmData[i][j],cmData[i][j]);
  	// 		}else{
  	// 			alert('存在同名行/列，数据将不准确!');
  	// 			return;
  	// 		}
  	// 	}
  	// }
  	// 生成各行/列的单元格占的span长度的数组
  	for(var i=0;i<rmData.length;i++){
  		d.rmLenArr.push(1);
  		for(var j=i+1;j<rmData.length;j++){
  			d.rmLenArr[i]=d.rmLenArr[i]*rmData[j].length;
  		}
  	}
  	for(var i=0;i<cmData.length;i++){
  		d.cmLenArr.push(1);
  		for(var j=i+1;j<cmData.length;j++){
  			d.cmLenArr[i]=d.cmLenArr[i]*cmData[j].length;
  		}
  	}
  	d.rmData=rmData;
  	d.cmData=cmData;
  	d.rmCount=rmData.length;
  	d.cmCount=cmData.length;
  	
  	// 头+各行列+汇总
  	var rTotalNum=d.rmLenArr[0]*d.rmData[0].length;
     	var cTotalNum=d.cmLenArr[0]*d.cmData[0].length;
  	d.tempRmData=new Array(d.rmCount+cTotalNum+1);
  	d.tempCmData=new Array(d.cmCount+rTotalNum+1);

		var tHtml='<table>';
		for(var i=0;i<d.tempCmData.length;i++){
			tHtml=tHtml+'<tr>';
			for(var j=0;j<d.tempRmData.length;j++){
				// 不含合计的部分
				if(i!=d.tempCmData.length-1 && j!=d.tempRmData.length-1){
					// 空白表头
					if(i==0 && j==0){
						tHtml=tHtml+'<td rowspan="'+d.cmCount+'" colspan="'+d.rmCount+'"></td>';
					}
					// 行头
					if(j>=d.rmCount && (j-d.rmCount)%d.cmLenArr[i]==0 && i<d.cmCount){
						var colspan=d.cmLenArr[i];
						var id=getId('colHead',j-d.rmCount,i)+"head-";
						tHtml=tHtml+'<td colspan="'+colspan+'" id="'+id+'">'+d.cmData[i][((j-d.rmCount)/d.cmLenArr[i]%d.cmData[i].length)]+'</td>';
					}
					// 列头
					if(i>=d.cmCount && (i-d.cmCount)%d.rmLenArr[j]==0 && j<d.rmCount){
						var rowspan=d.rmLenArr[j];
						var id="head-"+getId('rowHead',i-d.cmCount,j);
						tHtml=tHtml+'<td rowspan="'+rowspan+'" id="'+id+'">'+d.rmData[j][((i-d.cmCount)/d.rmLenArr[j]%d.rmData[j].length)]+'</td>';
					}
					// 待填鸭的非合计单元格
					if(i>=d.cmCount && j>=d.rmCount){
						var id=getId('all',i-d.cmCount,j-d.rmCount);
						tHtml=tHtml+'<td style="text-align: center" id="'+id+'"></td>';
					}
					// 合计部分
				}else{
					if(i==d.tempCmData.length-1){
						if(j==0){
							// 行合计头
							tHtml=tHtml+'<td colspan="'+d.rmCount+'">合计</td>';
							// 待填鸭的行合计单元格
						}else if(j< d.tempRmData.length-1 && j>=d.rmCount){
							var id='';
							for(var k=0;k<d.cmData.length;k++){
    							id=id+d.cmData[k][parseInt((j-d.rmCount)/d.cmLenArr[k]%d.cmData[k].length)]+'-';
    						}
    						tHtml=tHtml+'<td id="'+id+'"></td>';
    						// 总合计
						}else if(j==d.tempRmData.length-1){
							tHtml=tHtml+'<td id="total"></td>';
						}
					}else{
						// 列合计头
						if(i==0){
							tHtml=tHtml+'<td rowspan="'+d.cmCount+'">合计</td>';
							// 待填鸭的列合计单元格
						}else if(i < d.tempCmData.length-1 && i>=d.cmCount){
							var id='';
							for(var k=0;k<d.rmData.length;k++){
    							id=id+d.rmData[k][parseInt((i-d.cmCount)/d.rmLenArr[k]%d.rmData[k].length)]+'-';
    						}
    						tHtml=tHtml+'<td id="'+id+'"></td>';
						}
					}
				}
			}
			tHtml=tHtml+'</tr>';
		}
		tHtml=tHtml+'</table>';
		// console.log(tHtml);
		$("#root").html(tHtml);


      // 填入数值
      var mCount=d.rmCount+d.cmCount;
      for(var i=0;i<d.data.length;i++){
      	var id='#';
      	for(var j=0;j<d.data[i].length;j++){
      		if(j<mCount){
      			id=id+d.data[i][j]+'-';
      		}else{
      			$(id).html(d.data[i][j]);
      		}
      	}
      }
      // 计算合计
     	
     	var total=0;
     	for(var i=0;i<rTotalNum;i++){
     		// 获取合计的id
     		var id="";
     		for(var j=0;j<d.rmCount;j++){
     			id=id+rmData[j][parseInt(i/d.rmLenArr[j]%d.rmData[j].length)]+'-';
     		}
     		var arr=$("td[id^='"+id+"']");
     		var count=0;
     		for(var j=0;j<arr.length;j++){
     			var num=$(arr[j]).html();
     			if(num){
     				count=parseInt(count)+parseInt(num);
     			}
     		}
     		if(count){
     			$('#'+id).html(count);
     		}
     		total+=count;
     	}
     	$('#total').html(total);
     	for(var i=0;i<cTotalNum;i++){
     		// 获取合计的id
     		var id="";
     		for(var j=0;j<d.cmCount;j++){
     			id=id+cmData[j][parseInt(i/d.cmLenArr[j]%d.cmData[j].length)]+'-';
     		}
     		var arr=$("td[id$='"+id+"']");
     		var count=0;
     		for(var j=0;j<arr.length;j++){
     			var num=$(arr[j]).html();
     			if(num){
     				count=parseInt(count)+parseInt(num);
     			}
     		}
     		if(count){
     			$('#'+id).html(count);
     		}
     	}
     	// 隐藏整列没有数据的单元格
     	for(var i=0;i<cTotalNum;i++){
     		for(var j=0;j<rTotalNum;j++){
     			var id=getId('all',j,i);
				var val=$('#'+id).html();
				if(val){
					break;
				}
				if(j==rTotalNum-1){
					var toHideIdStr=getId('col',i);
					var arr=$("td[id$='"+toHideIdStr+"']");
					// 隐藏数值单元格
  	       		for(var k=0;k<arr.length;k++){
  	       			$(arr[k]).hide();
  	       		}
  	       		// 隐藏列头
  	       		var toHideIdArr=toHideIdStr.split('-');
  	       		var headId="";
  	       		for(var k=0;k<toHideIdArr.length;k++){
  	       			headId=headId+toHideIdArr[k]+"-";
  	       			var tempId='#'+headId+'head-';
  	       			var colspan=$(tempId).attr('colspan');
  	       			if(colspan && colspan<=1){
  	       				$(tempId).hide();
  	       			}else if(colspan && colspan>1){
  	       				$(tempId).attr('colspan',colspan-1);
  	       			}
  	       		}
				}
     		}
     	}

     	// 隐藏整行没有数据的单元格
     	for(var i=0;i<rTotalNum;i++){
     		for(var j=0;j<cTotalNum;j++){
     			var id=getId('all',i,j);
				var val=$('#'+id).html();
				if(val){
					break;
				}
				if(j==cTotalNum-1){
					var toHideIdStr=getId('row',i);
					var arr=$("td[id^='"+toHideIdStr+"']");
					// 隐藏数值单元格
  	       		for(var k=0;k<arr.length;k++){
  	       			$(arr[k]).hide();
  	       		}
  	       		// 行隐藏的话会自动调整rowspan!?因此只要隐藏最后一个行头
  	       		$('#head-'+toHideIdStr).hide();
  	       		// 隐藏行头
				}
     		}
     	}

     	function getId(type,t1,t2){
        var id="";
        if(type=='col'){
          for(var k=0;k<d.cmData.length;k++){
              id=id+d.cmData[k][parseInt(t1/d.cmLenArr[k]%d.cmData[k].length)]+'-';
            }
        }else if(type=='colHead'){
          for(var k=0;k<d.cmData.length && k<=t2;k++){
              id=id+d.cmData[k][parseInt(t1/d.cmLenArr[k]%d.cmData[k].length)]+'-';
            }
        }else if(type=='row'){
          for(var k=0;k<d.rmData.length;k++){
              id=id+d.rmData[k][parseInt(t1/d.rmLenArr[k]%d.rmData[k].length)]+'-';
            }
        }else if(type=='rowHead'){
          for(var k=0;k<d.rmData.length && k<=t2;k++){
              id=id+d.rmData[k][parseInt(t1/d.rmLenArr[k]%d.rmData[k].length)]+'-';
            }
        }else if(type=='all'){
          for(var k=0;k<d.rmData.length;k++){
              id=id+d.rmData[k][parseInt(t1/d.rmLenArr[k]%d.rmData[k].length)]+'-';
            }
            for(var k=0;k<d.cmData.length;k++){
              id=id+d.cmData[k][parseInt(t2/d.cmLenArr[k]%d.cmData[k].length)]+'-';
            }
        }
        return id;
  	    }
  });

  

  </script>
